try

  if verLessThan('matlab','8.3.0.532')
    nthreads = SetUpMatlabPool();
  else
    nthreads = SetUpParPool();
  end

catch ME

  warning('Could not set up matlabpool: %s',getReport(ME));
  nthreads = struct;

end
% Start JAABA.
